✅ 1. EXACT IRR FUNCTION (Newton-Raphson — stable)
def compute_irr(cashflows, guess=0.1, tol=1e-6, max_iter=1000):
    """
    Compute IRR using Newton-Raphson method

    cashflows: list of floats
        Example: [+3500, -5000] OR [-3500, +5000] depending on perspective
    """
    rate = guess

    for _ in range(max_iter):
        # NPV
        npv = 0.0
        d_npv = 0.0

        for t, cf in enumerate(cashflows):
            npv += cf / (1 + rate) ** t
            if t != 0:
                d_npv -= t * cf / (1 + rate) ** (t + 1)

        if abs(npv) < tol:
            return rate

        if d_npv == 0:
            raise ValueError("Derivative zero — IRR not solvable")

        rate -= npv / d_npv

    raise ValueError("IRR did not converge")

👉 This is based on solving NPV = 0 iteratively (how Excel/IRR works internally)

If your periods are daily / monthly, you MUST annualize:

def annualize_irr(periodic_irr, periods_per_year):
    return (1 + periodic_irr) ** periods_per_year - 1
✅ 3. FULL APR ENGINE (LoanLens-ready)
def calculate_apr(
    disbursed_amount,
    repayment_amount,
    tenure_days
):
    """
    Returns APR using IRR logic
    """

    # Borrower's perspective:
    # + money received, - money paid
    cashflows = [
        disbursed_amount,     # t=0 (positive inflow)
        -repayment_amount     # t=end (outflow)
    ]

    # Step 1: IRR per period
    irr = compute_irr(cashflows)

    # Step 2: annualize (daily compounding)
    apr = annualize_irr(irr, 365 / tenure_days)

    return apr

Irregular payments (XIRR-style)

Use time-weighted discounting:

def compute_xirr(cashflows, days):
    def npv(rate):
        return sum(cf / (1 + rate) ** (d / 365) for cf, d in zip(cashflows, days))

    rate = 0.1
    for _ in range(1000):
        f = npv(rate)
        df = sum(
            - (d / 365) * cf / (1 + rate) ** ((d / 365) + 1)
            for cf, d in zip(cashflows, days)
        )
        rate -= f / df

    return rate

class LoanValidationEngine:
    def __init__(self, data):
        self.data = data
        self.errors = []
        self.warnings = []
        self.flags = []

    # ----------------------------
    # BASIC SANITY CHECKS
    # ----------------------------
    def validate_numeric_fields(self):
        numeric_fields = [
            "principal_sanctioned",
            "amount_disbursed",
            "processing_fee",
            "gst_on_fee",
            "interest_rate_annual_percent",
            "tenure_days",
            "total_repayment"
        ]

        for field in numeric_fields:
            val = self.data.get(field)
            if val is not None:
                if not isinstance(val, (int, float)):
                    self.errors.append(f"{field} not numeric")
                elif val < 0:
                    self.errors.append(f"{field} negative")

    # ----------------------------
    # REQUIRED FIELD CHECKS
    # ----------------------------
    def validate_required_fields(self):
        required_any = [
            ("principal_sanctioned", "amount_disbursed"),
            ("total_repayment",),
        ]

        for group in required_any:
            if not any(self.data.get(f) for f in group):
                self.errors.append(f"Missing required fields: {group}")

    # ----------------------------
    # LOGICAL CONSISTENCY
    # ----------------------------
    def validate_consistency(self):
        p = self.data.get("principal_sanctioned")
        d = self.data.get("amount_disbursed")
        r = self.data.get("total_repayment")

        # repayment should be >= disbursed
        if d and r and r < d:
            self.errors.append("Repayment less than disbursed (invalid loan)")

        # disbursed > sanctioned (impossible)
        if p and d and d > p:
            self.errors.append("Disbursed greater than sanctioned")

    # ----------------------------
    # FEE VALIDATION
    # ----------------------------
    def validate_fees(self):
        p = self.data.get("principal_sanctioned")
        fee = self.data.get("processing_fee")

        if p and fee:
            if fee > p:
                self.errors.append("Processing fee exceeds principal")

            # suspicious high fee
            if fee / p > 0.5:
                self.flags.append("HIGH_PROCESSING_FEE")

    # ----------------------------
    # GST VALIDATION
    # ----------------------------
    def validate_gst(self):
        fee = self.data.get("processing_fee")
        gst = self.data.get("gst_on_fee")

        if fee and gst:
            expected_gst = fee * 0.18

            if abs(gst - expected_gst) > fee * 0.05:
                self.warnings.append("GST mismatch (not ~18%)")

    # ----------------------------
    # INTEREST RATE VALIDATION
    # ----------------------------
    def validate_interest_rate(self):
        rate = self.data.get("interest_rate_annual_percent")

        if rate:
            if rate < 0 or rate > 200:
                self.flags.append("EXTREME_INTEREST_RATE")

    # ----------------------------
    # TENURE VALIDATION
    # ----------------------------
    def validate_tenure(self):
        t = self.data.get("tenure_days")

        if t:
            if t <= 0:
                self.errors.append("Invalid tenure")

            if t < 7:
                self.flags.append("ULTRA_SHORT_LOAN")

    # ----------------------------
    # FEE DEDUCTION TRAP
    # ----------------------------
    def detect_fee_trap(self):
        p = self.data.get("principal_sanctioned")
        d = self.data.get("amount_disbursed")

        if p and d and d < p:
            self.flags.append("FEE_DEDUCTION_TRAP")

    # ----------------------------
    # APR CONSISTENCY CHECK
    # ----------------------------
    def validate_apr(self, computed_apr):
        declared = self.data.get("apr_declared_percent")

        if declared and computed_apr:
            if abs(declared - computed_apr * 100) > 5:
                self.flags.append("APR_MISMATCH")

    # ----------------------------
    # PENAL CHARGE VALIDATION
    # ----------------------------
    def validate_penalties(self):
        penalties = self.data.get("penal_charges") or []

        for p in penalties:
            rate = p.get("rate_percent")
            if rate and rate > 10:
                self.flags.append("HIGH_PENALTY_RATE")

    # ----------------------------
    # FINAL RUNNER
    # ----------------------------
    def run_all(self, computed_apr=None):
        self.validate_numeric_fields()
        self.validate_required_fields()
        self.validate_consistency()
        self.validate_fees()
        self.validate_gst()
        self.validate_interest_rate()
        self.validate_tenure()
        self.detect_fee_trap()
        self.validate_penalties()

        if computed_apr:
            self.validate_apr(computed_apr)

        return {
            "cleaned_data": self.data,
            "errors": self.errors,
            "warnings": self.warnings,
            "flags": self.flags,
            "is_valid": len(self.errors) == 0
        }
